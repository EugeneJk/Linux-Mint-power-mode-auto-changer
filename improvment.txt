ACTIVE_USER=$(loginctl list-sessions --no-legend | awk '{print $3}' | head -n1)
ACTIVE_UID=$(id -u "$ACTIVE_USER")

DBUS_SESSION_BUS_ADDRESS="unix:path=/run/user/$ACTIVE_UID/bus"

sudo -u "$ACTIVE_USER" \
DBUS_SESSION_BUS_ADDRESS=$DBUS_SESSION_BUS_ADDRESS \
gdbus call --session \
  --dest org.Cinnamon \
  --object-path /org/Cinnamon \
  --method org.Cinnamon.ShowOSD \
  "{'icon': <'ac-adapter-symbolic'>, 'label': <'AC mode activated'>}"



notify-send -i battery-level-50-symbolic \
"Battery level 50%."


notify-send -i battery-low-symbolic \
"Battery level 20%."
